const crearVenta = async (req, res) => {
    const { cedula_cc, nombres_completos, correo_electronico, numero_motor, modelo, precio_venta } = req.body;

    // 1. Validaciones manuales directas
    if (!cedula_cc || !nombres_completos || !modelo) {
        return res.status(400).json({
            ok: false,
            mensaje: "Faltan datos obligatorios (Cédula, Nombre o Modelo)"
        });
    }

    if (typeof precio_venta !== 'number' || precio_venta <= 0) {
        return res.status(400).json({
            ok: false,
            mensaje: "El precio debe ser un número válido mayor a 0"
        });
    }

    try {
        // 2. Insertar en BD
        const nuevaVenta = await VentaModel.create({ 
            cedula_cc, nombres_completos, correo_electronico, numero_motor, modelo, precio_venta 
        });

        // 3. Respuesta exitosa al Front
        return res.status(201).json({
            ok: true,
            mensaje: "Venta guardada correctamente",
            venta_id: nuevaVenta._id // o nuevaVenta.id
        });

    } catch (error) {
        return res.status(500).json({
            ok: false,
            mensaje: "Error interno al guardar",
            detalle: error.message
        });
    }
};
